```typescript
export class PromptRefiner {
  static async refinePromptWithFeedback(
    originalPrompt: string,
    feedback: string
  ): Promise<string> {
    const response = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: [
        {
          role: "system",
          content: `You are an expert at refining video editing prompts based on user feedback.`
        },
        {
          role: "user",
          content: `Original prompt: "${originalPrompt}"\n\nUser feedback: "${feedback}"`
        }
      ],
      temperature: 0.7,
      max_tokens: 300,
    });
    
    return response.choices[0]?.message?.content?.trim() || originalPrompt;
  }
}
```

### Features:
- ✅ Uses GPT-4o
- ✅ Takes original prompt + feedback
- ✅ Returns refined prompt
- ✅ Error handling (returns original on failure)
- ✅ Bonus: Batch refine support
